    - Create private method `performPostLoopGitActions(loopId: string)`:
      ```typescript
      private async performPostLoopGitActions(loopId: string): Promise<void> {
        const run = await this.loopRuns.findById(loopId)
        if (!run?.branchName || !run.baseBranch) return

        const project = await this.projects.findById(run.projectId)
        if (!project) return

        const runCwd = run.worktree
          ? (await this.diffService.resolveWorktreePath(project.path, run.worktree)) ?? project.path
          : project.path

        // Commit any uncommitted changes
        if (await this.gitService!.hasPendingChanges(runCwd)) {
          await this.gitService!.commitAllChanges(runCwd, `Loop ${loopId.slice(0, 8)} final changes`)
        }

        // Push branch
        const pushResult = await this.gitService!.pushBranch(runCwd, run.branchName)
        if (!pushResult.pushed) {
          console.warn(`[LoopService] Push failed for loop ${loopId}: ${pushResult.reason}`)
          return
        }

        // Create PR if GitHub info available
        if (!project.githubOwner || !project.githubRepo || !project.userId) return
        if (!this.githubService) return

        let githubToken: string
        try {
          githubToken = await this.githubService.getDecryptedToken(project.userId)
        } catch {
          console.warn(`[LoopService] No GitHub token for user ${project.userId}, skipping PR`)
          return
        }

        const prTitle = `Loop ${loopId.slice(0, 8)}: ${run.prompt?.slice(0, 60) ?? 'Automated changes'}`
        const prBody = [
          `Automated PR from Ralph loop \`${loopId}\`.`,
          '',
          run.prompt ? `**Prompt:** ${run.prompt.slice(0, 200)}` : '',
          `**Iterations:** ${run.iterations}`,
          `**Tokens used:** ${run.tokensUsed}`
        ].filter(Boolean).join('\n')

        const prResult = await this.gitService!.createPullRequest({
          cwd: runCwd,
          branchName: run.branchName,
          baseBranch: run.baseBranch,
          title: prTitle,
          body: prBody,
          githubToken,
          owner: project.githubOwner,
          repo: project.githubRepo
        })

        if (prResult.prUrl) {
          await this.loopRuns.update(loopId, { prUrl: prResult.prUrl })
          this.events.emit(`${STATE_EVENT_PREFIX}${loopId}`, 'completed')
          console.log(`[LoopService] PR created for loop ${loopId}: ${prResult.prUrl}`)
        } else {
          console.warn(`[LoopService] PR creation failed for loop ${loopId}: ${prResult.reason}`)
        }
      }
      ```
